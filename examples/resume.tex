% Macros for the CV/resume entries.

% Section.
% Takes the name of the section and stylizes it, adding a
% horizontal line after it.
\newcommand{\resumesection}[1]{
    {
        \color{accentcolor}
        \noindent
        \large{#1}
        \vspace{-8pt}

        \noindent
        \rule{\textwidth}{2pt}
    }
}

% Personal information entry.
% Takes a pair of facts about you as input. They can be
% your website, your social media, your email,
% They get put in one line, opposite to each other.
% and so on.
\newcommand{\info}[2]{
    {
        \color{secondarytext}
        \trivlist\nopagebreak
        \parindent0pt
        \item #1 \hspace{\fill} #2
        \item\relax\obeylines
    }
    \vspace{-5pt}
}

% Experience/extracurricular activity entry.
% Takes the following:
% 1. Date (in whatever format you want).
% 2. Title or position.
% 3. Company, organization or institution.
% 4. Place.
% 5. Description (in whatever format you like).
\newcommand{\job}[5]{
    \textbf{#2}, #3 \hspace{\fill} #1

    {\color{secondarytext}#4}

    #5

    \vspace{-5pt}
    {
        \color{secondaccentcolor}
        \rule{0.96\textwidth}{0.5pt}
    }

    \vspace{15pt}
}

% Education entry.
% Takes the following:
% 1. Date (in whatever format you want).
% 2. Degree.
% 3. Institution.
% 4. Place.
\newcommand{\education}[4]{
    \textbf{#2}, #3 \hspace{\fill} #1

    {\color{secondarytext}#4}

    \vspace{-5pt}
    {
        \color{secondaccentcolor}
        \rule{0.96\textwidth}{0.5pt}
    }

    \vspace{15pt}
}

% Project entry.
% Takes the following:
% 1. Project name.
% 2. Link to the project.
% 3. Description (again, in whatever format you want).
\newcommand{\project}[3]{
    #1 \hspace{\fill} \url{#2}

    #3

    \vspace{-5pt}
    {
        \color{secondaccentcolor}
        \rule{0.96\textwidth}{0.5pt}
    }

    \vspace{15pt}
}

% Language entry.
% You can opt to not use this entry, it just to make a
% pretty listing of the languages you know.
% Takes the following:
% 1. Language.
% 2. Proficiency level.
\newcommand{\languageprof}[2]{
    \raisebox{3pt}{$\centerdot$} \textbf{#1} - #2

    \vspace{5pt}
}

% Award/honor entry. If you change the font size, you'll
% probably have to change the hspace inside this macro.
% Takes the following:
% 1. Name of the award/honor.
% 2. Year.
% 3. Description of the award/honor (optional).
\newcommand{\award}[3]{
    \raisebox{3pt}{$\centerdot$} #1 \hspace{\fill} #2

    \ifthenelse{\equal{#3}{}}{
    }{
        \hspace{7pt}#3
    }

    \vspace{5pt}
}

% List entry.
% Makes bullet points that have less indentation and less
% space between lines.
\newcommand{\closerlist}[1]{
    \raisebox{3pt}{$\centerdot$} #1

    \vspace{5pt}
}
